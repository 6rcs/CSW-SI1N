            import pandas as pd
import numpy as np

# Exemplo de conjunto de dados
data = {'coluna1': [10, 12, 14, 100, 15, 16, 19, 18, 20, 25, 200]}
df = pd.DataFrame(data)

# Cálculo do IQR (Intervalo Interquartil)
Q1 = df['coluna1'].quantile(0.25)
Q3 = df['coluna1'].quantile(0.75)
IQR = Q3 - Q1

# Definindo limites para os outliers
limite_inferior = Q1 - 1.5 * IQR
limite_superior = Q3 + 1.5 * IQR

# Removendo os outliers
df_sem_outliers = df[(df['coluna1'] >= limite_inferior) & (df['coluna1'] <= limite_superior)]

print("Dataframe original:")
print(df)
print("\nDataframe sem outliers:")
print(df_sem_outliers)
